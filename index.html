إنَّ الْحَمْدَ لِلهِ نَحْمَدُهُ وَنَسْتَعِينُهُ وَنَسْتَغْفِرُهُ، وَنَعُوذُ بِاللهِ مِنْ شُرُورِ أَنْفُسِنَا وَمِنْ سَيِّئَاتِ أَعْمَالِنَا، مَنْ يَهْدِهِ اللهُ فَلاَ مُضِلَّ لَهُ، وَمَنْ يُضْلِلْ فَلاَ هَادِىَ لَهُ، وَأَشْهَدُ أَنْ لاَ إِلَهَ إِلاَّ اللهُ وَحْدَهُ لاَ شَرِيكَ لَهُ، وَأَشْهَدُ أَنَّ مُحَمَّدًا عَبْدُهُ وَرَسُولُهُ
عُمَرَ بْنَ الْخَطَّابِ رَضِيَ اللَّهُ عَنْهُ قَالَ : سَمِعْتُ رَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ، يَقُولُ : " إِنَّمَا الْأَعْمَالُ بِالنِّيَّاتِ، وَإِنَّمَا لِكُلِّ امْرِئٍ مَا نَوَى، فَمَنْ كَانَتْ هِجْرَتُهُ إِلَى دُنْيَا يُصِيبُهَا أَوْ إِلَى امْرَأَةٍ يَنْكِحُهَا، فَهِجْرَتُهُ إِلَى مَا هَاجَرَ إِلَيْهِ "
صحيح البخاري
ޢުމަރު ބުން އަލްޚައްޠާބު رَضِيَ اللَّهُ عَنْهُގެ އަރިހުން ރިވާވެގެންވިއެވެ. ރަސޫލު الله ﷺ ޙަދީޘްކުރައްވަނިކޮށް ތިމަންކަލޭގެފާނު އަޑުއެއްސެވީމެވެ. ”ނިޔަތްގަތް ގޮތުގެ މައްޗަށް މެނުވީ ޢަމަލުތައް ބިނާވެގެން ނުވެއެވެ. އަދި ކޮންމެ މީހަކަށްވެސް ވަނީ އެމީހަކު ނިޔަތްގަތް ކަމެކެވެ. ފަހެ، الله އާއި އެއިލާހުގެ ރަސޫލާއަށް ޓަކައި ހިޖުރަކޮއްފި މީހާގެ ހިޖުރަ ވާހުށީ الله އާއި އެއިލާހުގެ ރަސޫލާއަށް ޓަކައި ކުރެވުނު ހިޖުރައެއްގެ ގޮތުގައެވެ. އަދި ދުނިޔަވީ ކަމަކަށްޓަކައި ނުވަތަ މީހަކާ ކައިވެނި ކުރުމަށްޓަކައި ހިޖުރަކުރި މީހާގެ ހިޖުރަ ވާހުށީ އޭނާ ހިޖުރަކުރި ކަމެއްގެ މައްޗަށެވެ.“
وقال تعالى:
‎﴿ وَلَقَدْ أُوحِىَ إِلَيْكَ وَإِلَى ٱلَّذِينَ مِن قَبْلِكَ لَئِنْ أَشْرَكْتَ لَيَحْبَطَنَّ عَمَلُكَ وَلَتَكُونَنَّ مِنَ ٱلْخَـٰسِرِينَ ﴾‏
سورة الزمر 65
الله تعالى ވަޙީ ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން ކަލޭގެފާނާށާއި، ކަލޭގެފާނުގެ ކުރީގެ ރަސޫލުންނަށް ވަޙީ ކުރައްވާފައިވެއެވެ. ކަލޭގެފާނު ޝަރީކު ކުރައްވާފިނަމަ، ކަލޭގެފާނުގެ އަމަލުތައް ބާތިލު ވެއްޖައީސްކަން ކަށަވަރެވެ. އަދި އޭރުން ކަލޭގެފާނު ވާހުށީ ގެއްލެނިވެގެންވާ މީހުންގެ ތެރެއިންކަން ކަށަވަރެވެ.“
وقال تعالى:
‎﴿ ... وَلَوْ أَشْرَكُوا۟ لَحَبِطَ عَنْهُم مَّا كَانُوا۟ يَعْمَلُونَ ﴾‏
سورة الأنعام 88
الله تعالى ވަޙީ ކުރެއްވިއެވެ: ”އެ ބޭކަލުންވެސް (الله އަށް) ޝަރީކުުރެއްވިނަމަ. އެ ބޭކަލުން ކޮށްއުޅުއްވި އަމަލުތައް ބާތިލުވީހެވެ.“
قال الله تعالى:
‎﴿ ... إِنَّهُۥ مَن يُشْرِكْ بِٱللَّهِ فَقَدْ حَرَّمَ ٱللَّهُ عَلَيْهِ ٱلْجَنَّةَ وَمَأْوَىٰهُ ٱلنَّارُ ۖ وَمَا لِلظَّـٰلِمِينَ مِنْ أَنصَارٍۢ ﴾‏
سورة المائدة 72
الله تعالى ވަޙީ ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން الله އަށް ޝަރީކުކުރާ މީހާ (ދަންނާށެވެ!) ފަހެ ހަމަކަށަވަރުން الله، އެ މީހާގެ މައްޗަށް ސުވަރުގެ ހަރާމްކުރެއްވިއެވެ. އަދި އެ މީހަކު އުޅޭނެތަނަކީ ނަރަކައެވެ. އަދި އަނިޔާވެރިންނަށް (އެބަހީ: ޝަރީކުކުރި މީހުންނަށް) ނަސްރު ދޭނޭ އެއްވެސް ބަޔަކު ނުވެއެވެ.“
وقال تعالى:
‎﴿ إِنَّ ٱللَّهَ لَا يَغْفِرُ أَن يُشْرَكَ بِهِۦ وَيَغْفِرُ مَا دُونَ ذَٰلِكَ لِمَن يَشَآءُ ۚ وَمَن يُشْرِكْ بِٱللَّهِ فَقَدِ ٱفْتَرَىٰٓ إِثْمًا عَظِيمًا ﴾‏
سورة النساء 48
الله تعالى ވަޙީ ކުރެއްވިއެވެ: ”ހަމަކަށަވަރުން الله، އެ އިލާހަށް ޝަރީކު ކުރެވުމަށް ނުފުއްސަވައެވެ. އަދި އެ ނޫން ފާފަތައް، އެ އިލާހު އިރާދަ ކުރައްވާ މީހަކަށް ފުއްސަވައެވެ. އަދި الله އަށް ޝަރީކު ކުރާ މީހާ، ފަހެ ބޮޑުވެގެންވާ ދޮގު ފާފައެއް އުފައްދައިފިކަން ކަށަވަރެވެ.“
وقال تعالى:
‎﴿ وَمَآ أَرْسَلْنَا مِن قَبْلِكَ مِن رَّسُولٍ إِلَّا نُوحِىٓ إِلَيْهِ أَنَّهُۥ لَآ إِلَـٰهَ إِلَّآ أَنَا۠ فَٱعْبُدُونِ ﴾‏
سورة الأنبياء 25
الله تعالى ވަޙީ ކުރެއްވިއެވެ: ”އަދި ހަމަކަށަވަރުން ތިމަން އިލާހު މެނުވީ (އަޅުކަންކުރުން ހައްގުވާ) އިލާހަކު ނުވާކަމަށް އެ ބޭކަލަކަށް ވަހީ ނުކުރައްވާ އެއްވެސް ރަސޫލަކު، ކަލާގެފާނުގެ ކުރިންވެސް ތިމަން އިލާހު ނުފޮނުއްވަމެވެ. ފަހެ ތިޔަބައިމީހުން (ހަމައެކަނި) ތިމަން އިލާހަށް އަޅުކަން ކުރާށެވެ!“
وقال تعالى:
‎﴿ وَلَقَدْ بَعَثْنَا فِى كُلِّ أُمَّةٍۢ رَّسُولًا أَنِ ٱعْبُدُوا۟ ٱللَّهَ وَٱجْتَنِبُوا۟ ٱلطَّـٰغُوتَ ۖ... ﴾‏
سورة النحل 36
الله تعالى ވަޙީ ކުރެއްވިއެވެ: ”އަދި ހަމަކަށަވަރުން ތިމަން އިލާހު ކޮންމެ އުއްމަތަކަށްމެ ރަސޫލަކު ފޮނުއްވީމެވެ. އެއީ اللهއަށް އަޅުކަން ކުރުމަށާއި، ޠާޣޫތުން ދުރުވެގަތުމަށް އެންގެވުމަށެވެ.“
